FinalUi

"use client";
import React, { useState } from "react";
import { Globe2 } from "lucide-react";
import { Button } from "@/components/ui/button";
import axios from "axios";

interface FinalUiProps {
  viewTrip?: () => void;
  disable?: boolean;
  origin?: string;
  destination?: string;
  groupSize?: string;
  budget?: string;
  duration?: string;
}


function FinalUi({ origin, destination, groupSize, budget, duration }: FinalUiProps) {
  const [tripData, setTripData] = useState<any>(null);
  const [loading, setLoading] = useState(false);

  const generateTrip = async () => {
    setLoading(true);
    try {
      const response = await axios.post("/api/trip", {
        messages: [
          {
            role: "user",
            content: `${origin} to ${destination}, group: ${groupSize}, budget: ${budget}, duration: ${duration}`,
          },
        ],
        isFinal: true,
      });

      setTripData(response.data);
      setLoading(false);
    } catch (error) {
      console.error("Trip generation failed:", error);
      setLoading(false);
    }
  };

  const viewTrip = () => {
    // Could navigate to another page or expand view in-place
    window.scrollTo({ top: 0, behavior: "smooth" });
  };

  return (
    <div className="flex flex-col items-center justify-center mt-6 p-6 bg-white rounded-2xl shadow-md">
      {!tripData ? (
        <>
          <Globe2 className="text-primary text-4xl animate-bounce" />
          <h2 className="mt-3 text-lg font-semibold text-primary text-center">
            Planning your dream trip...
          </h2>
          <p className="text-gray-500 text-sm text-center mt-1">
            Gathering best destinations, activities, and travel details for you.
          </p>

          {/* Progress Bar */}
          <div className="w-48 h-2 bg-gray-200 rounded-full mt-4 overflow-hidden">
            <div
              className="h-2 bg-primary animate-pulse"
              style={{ width: "75%" }}
            ></div>
          </div>

          <Button
            disabled={true}
            className="mt-4 w-full cursor-not-allowed"
          >
            View Trip
          </Button>

          <Button
            onClick={generateTrip}
            className="mt-4 w-full bg-primary text-white"
          >
            Generate Trip
          </Button>
        </>
      ) : (
        <>
          <h2 className="text-xl font-bold text-center text-primary mb-4">
            Your Trip Plan is Ready!
          </h2>

          {/* Hotels */}
          {tripData.trip_plan?.hotels?.map((hotel: any, idx: number) => (
            <div
              key={idx}
              className="mb-4 p-4 border rounded-xl shadow-sm w-full max-w-xl"
            >
              <h3 className="font-semibold text-lg">{hotel.hotel_name}</h3>
              <p>{hotel.hotel_address}</p>
              <p>Price per night: {hotel.price_per_night}</p>
              <p>Rating: {hotel.rating}</p>
              <p>{hotel.description}</p>
              <img
                src={hotel.hotel_image_url}
                alt={hotel.hotel_name}
                className="mt-2 w-full h-48 object-cover rounded-lg"
              />
            </div>
          ))}

          {/* Itinerary */}
          {tripData.trip_plan?.hotels?.map((hotel: any, idx: number) =>
            hotel.itinerary?.map((day: any) => (
              <div
                key={`${idx}-${day.day}`}
                className="mb-4 p-4 border rounded-xl shadow-sm w-full max-w-xl"
              >
                <h4 className="font-semibold text-md">Day {day.day}</h4>
                <p>{day.day_plan}</p>
                <p>Best Time to Visit: {day.best_time_to_visit_day}</p>
                {day.activities?.map((activity: any, aidx: number) => (
                  <div
                    key={aidx}
                    className="mt-2 p-2 border rounded-lg bg-gray-50"
                  >
                    <h5 className="font-semibold">{activity.place_name}</h5>
                    <p>{activity["place details"]}</p>
                    <p>Address: {activity.place_address}</p>
                    <p>Ticket: {activity.ticket_pricing}</p>
                    <p>Travel Time: {activity.time_travel_each_location}</p>
                    <img
                      src={activity.place_image_url}
                      alt={activity.place_name}
                      className="mt-1 w-full h-32 object-cover rounded-lg"
                    />
                  </div>
                ))}
              </div>
            ))
          )}

          <Button
            onClick={viewTrip}
            className="mt-4 w-full bg-primary text-white"
          >
            View Trip
          </Button>
        </>
      )}
    </div>
  );
}

export default FinalUi;
